184  for(size_t k = 1;k <= this->max_iteration;k++) {
185  /* vk = (A + offset*E)uk, here E is the identity matrix */
186  std::fill(vk.begin(), vk.end(), 0.0);
187  this->mv_mul(uk, vk);
188  for(size_t i = 0;i < n;i++) {
189  vk[i] += uk[i]*this->eigenvalue_offset;

192  alphak = std::real(util::inner_prod(u.back(), vk));
193 
194  /* The inner product <uk|vk> is real.
195  * Proof:
196  * <uk|vk> = <uk|A|uk>
197  * On the other hand its complex conjugate is
198  * <uk|vk>^* = <vk|uk> = <uk|A^*|uk> = <uk|A|uk>
199  * here the condition that matrix A is a symmetric (Hermitian) is used.
200  * Therefore
201  * <uk|vk> = <vk|uk>^*
202  * <uk|vk> is real.
203  */
204 
205  alpha.push_back(alphak);
206 
207  for(size_t i = 0;i < n; i++) {
208  uk[i] = vk[i] - betak*u[k-1][i] - alphak*u[k][i];
209  }
